namespace User {

    namespace Entity {
        enum UserStatus {
            notConfirmed
            active
            inactive
        }

        class Anonymous {
            - //UUID//  **id**
            - //UUID//  **userId**
        }

        class User {
            - //UUID//  **id**
            - //String//   **firstName**
            - //String//   **lastName**
            - //String//   **email**
            - //UserStatus//   **status**

        }

        Anonymous --* User

        interface CredentialInterface
    }

    interface LoginHandler {
        + execute()
    }

    interface RegistrationHandler {
        + execute()
    }

    interface CredentialRepository {
        + save()
        + findById(id)
        + check(  //Entity.CredentialInterface//      ** cred** )
        + findByProviderId(     //providerId//    id)
    }
}
